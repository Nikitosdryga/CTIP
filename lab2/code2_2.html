Код:

                <body>
                <img id="movingImage" src="/1.jpg" alt="Зображення" />
            
                <script>
                    // отримую елемент зображення
                    const img = document.getElementById('movingImage');
            
                    // анімації
                    function moveImage() {
                        // розміри вікна
                        const windowWidth = window.innerWidth;
                        const windowHeight = window.innerHeight;
            
                        // рухаю зображення до правого нижнього кута
                        img.style.transform = `translate(${windowWidth - 100}px, ${windowHeight - 100}px)`;
                    }
            
                    // запуск анімацію через 1 секунду після завантаження сторінки
                    window.onload = function() {
                        setTimeout(moveImage, 1000);  // затримка 1 секунда
                    };
                </script>
            </body>
