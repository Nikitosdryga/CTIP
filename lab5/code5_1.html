Код:

                <body>
                <input type="text" id="textField" placeholder="Наведись на це поле">
                
                <div id="modal"></div>
            
                <script src="https://code.jquery.com/jquery-3.6.0.min.js"></script>
                <script>
                    $(document).ready(function() {
                        // наведення миші на текстове поле
                        $('#textField').hover(function() {
                            // отримане значення текстового поля
                            var value = $(this).val();
                            
                            // вивожу значення в модальне вікно
                            $('#modal').text(value).fadeIn();
                        }, function() {
                            // сховаю модальне вікно, коли курсор залишає поле
                            $('#modal').fadeOut();
                        });
                    });
                </script>
            </body>
