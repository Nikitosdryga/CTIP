DTD схема для "Інформація про пропозиції туристичної агенції"

<!DOCTYPE touristOffers [
  <!ELEMENT touristOffers (offer+)>
  
  <!ELEMENT offer (country, tourType, tourDescription, hotTour, price, quantity)>
  <!ATTLIST offer id ID #REQUIRED>
  
  <!ELEMENT country (#PCDATA)>
  <!ELEMENT tourType (#PCDATA)>
  <!ELEMENT tourDescription (#PCDATA)>
  <!ELEMENT hotTour (#PCDATA)>
  <!ELEMENT price (#PCDATA)>
  <!ELEMENT quantity (#PCDATA)>
  
]>

XML для "Інформація про пропозиції туристичної агенції"

<?xml version="1.0" encoding="UTF-8"?>
<xs:schema xmlns:xs="http://www.w3.org/2001/XMLSchema">

  <xs:element name="touristOffers">
    <xs:complexType>
      <xs:sequence>
        <xs:element name="offer" type="offerType" minOccurs="1" maxOccurs="unbounded"/>
      </xs:sequence>
    </xs:complexType>
  </xs:element>

  <xs:complexType name="offerType">
    <xs:sequence>
      <xs:element name="country" type="xs:string"/>
      <xs:element name="tourType" type="xs:string"/>
      <xs:element name="tourDescription" type="xs:string"/>
      <xs:element name="hotTour" type="xs:string"/>
      <xs:element name="price" type="xs:decimal"/>
      <xs:element name="quantity" type="xs:int"/>
    </xs:sequence>
    <xs:attribute name="id" type="xs:ID" use="required"/>
  </xs:complexType>

</xs:schema>
